import tkinter as tk

class GridLayoutDemo:
    def __init__(self, root):
        self.root = root
        self.root.title("Gridå¸ƒå±€ç¤ºä¾‹")
        
        # å‰µå»ºè¨ˆç®—å™¨ç•Œé¢
        self.create_calculator()
    
    def create_calculator(self):
        # é¡¯ç¤ºå€åŸŸ
        self.display = tk.Entry(self.root, width=20, font=('Arial', 14))
        self.display.grid(row=0, column=0, columnspan=4, pady=5)
        
        # æŒ‰éˆ•æ–‡æœ¬
        buttons = [
            '7', '8', '9', '/',
            '4', '5', '6', '*',
            '1', '2', '3', '-',
            '0', '.', '=', '+'
        ]
        
        # å‰µå»ºæŒ‰éˆ•
        row = 1
        col = 0
        for button in buttons:
            cmd = lambda x=button: self.click_button(x)
            tk.Button(self.root, text=button, width=5,
                     command=cmd).grid(row=row, column=col)
            col += 1
            if col > 3:
                col = 0
                row += 1
    
    def click_button(self, value):
        if value == '=':
            try:
                result = eval(self.display.get())
                self.display.delete(0, tk.END)
                self.display.insert(tk.END, str(result))
            except:
                self.display.delete(0, tk.END)
                self.display.insert(tk.END, "éŒ¯èª¤")
        else:
            self.display.insert(tk.END, value)

import tkinter as tk

    def __init__(self, root):
        self.root = root
        self.root.title("å¾…è¾¦äº‹é …")
        self.root.geometry("400x500")
        
        # æ•¸æ“šæ–‡ä»¶
        self.data_file = "todos.json"
        
        # åŠ è¼‰æ•¸æ“š

        # å‰µå»ºç•Œé¢

    def create_widgets(self):

        input_frame = ttk.Frame(self.root)

        self.task_entry = ttk.Entry(
            input_frame,
            textvariable=self.task_var,

        )

        ttk.Button(
            input_frame,
            text="æ·»åŠ ",
            command=self.add_task

        list_frame = ttk.Frame(self.root)

        self.task_list = ttk.Treeview(
            list_frame,
            columns=('Status',),
            show='tree'
        )
        self.task_list.pack(side='left', fill='both', expand=True)
        
        # æ»¾å‹•æ¢
        scrollbar = ttk.Scrollbar(
            list_frame,
            orient='vertical',
            command=self.task_list.yview

        scrollbar.pack(side='right', fill='y')
        
        self.task_list.configure(yscrollcommand=scrollbar.set)
        
        # ç¶å®šé›™æ“Šäº‹ä»¶
        self.task_list.bind('<Double-1>', self.toggle_task)
        
        # æ›´æ–°ä»»å‹™åˆ—è¡¨
        self.update_task_list()
    
    def load_data(self):
        """åŠ è¼‰æ•¸æ“š"""
        if os.path.exists(self.data_file):
            with open(self.data_file, 'r', encoding='utf-8') as f:
                self.tasks = json.load(f)
        else:
            self.tasks = []
    
    def save_data(self):
        """ä¿å­˜æ•¸æ“š"""
        with open(self.data_file, 'w', encoding='utf-8') as f:
            json.dump(self.tasks, f, ensure_ascii=False)
    
    def add_task(self):
        """æ·»åŠ ä»»å‹™"""
        task = self.task_var.get().strip()
        if task:
            self.tasks.append({
                'text': task,
                'done': False

            self.update_task_list()
            self.task_var.set('')
        else:
            messagebox.showwarning("è­¦å‘Š", "è«‹è¼¸å…¥ä»»å‹™å…§å®¹")
    
    def toggle_task(self, event):
        """åˆ‡æ›ä»»å‹™ç‹€æ…‹"""
        item = self.task_list.selection()[0]
        idx = self.task_list.index(item)
        self.tasks[idx]['done'] = not self.tasks[idx]['done']
        self.save_data()
        self.update_task_list()
    
    def update_task_list(self):
        """æ›´æ–°ä»»å‹™åˆ—è¡¨"""
        for item in self.task_list.get_children():
            self.task_list.delete(item)
        
        for task in self.tasks:
            status = 'âœ“' if task['done'] else ' '
            self.task_list.insert(
                '',
                'end',
                text=f"{status} {task['text']}"
            )

    root = tk.Tk()
    app = TodoApp(root)
    root.mainloop()
